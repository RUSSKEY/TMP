50{

53
54 for (const QString& rawCommand : commands) {

58 qDebug() << "Processing command:" << command;

60 QStringList parts = command.split(" ", Qt::SkipEmptyParts);
61 if (parts.isEmpty()) {

64 }
65
66 QString cmd = parts[0].toLower();

68
69 if (cmd == "reg") {
